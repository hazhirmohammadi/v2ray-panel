# We name the project and the target for the ExternalProject_Add() call
# to something that will highlight to the user what we are working on if
# something goes wrong and an error message is produced.

# paying the cost of redetecting them again in ExternalProject_Add()

  [==[C:/Program Files/Git/cmd/git.exe;2.40.0.windows.1]==]
)


include(ExternalProject)
ExternalProject_Add(asiocmake-populate
                     "UPDATE_DISCONNECTED" "False" "GIT_REPOSITORY" "https://github.com/OlivierLDff/asio.cmake" "GIT_TAG" "main"
                    SOURCE_DIR          "G:/c++/v2ray panel/backend/cmake-build-debug/_deps/asiocmake-src"
                    BINARY_DIR          "G:/c++/v2ray panel/backend/cmake-build-debug/_deps/asiocmake-build"
                    CONFIGURE_COMMAND   ""
                    BUILD_COMMAND       ""
                    INSTALL_COMMAND     ""
                    TEST_COMMAND        ""
                    USES_TERMINAL_DOWNLOAD  YES
                    USES_TERMINAL_UPDATE    YES
                    USES_TERMINAL_PATCH     YES
)
